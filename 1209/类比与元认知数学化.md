## ✅ Step 2：定义相似度（核心）

Python运行def graph_similarity(G1, G2):  
    # 简单版本：比较边结构  
    edges1 = set(G1.edges())  
    edges2 = set(G2.edges())  
      
    intersection = len(edges1 & edges2)  
    union = len(edges1 | edges2)  
      
    if union == 0:  
        return 0  
    return intersection / union

## ✅ Step 3：类比映射（关键）

Python运行import itertools  
  
def find_mapping(G1, G2):  
    nodes1 = list(G1.nodes())  
    nodes2 = list(G2.nodes())  
      
    best_score = 0  
    best_map = None  
      
    for perm in itertools.permutations(nodes2, len(nodes1)):  
        mapping = dict(zip(nodes1, perm))  
          
        mapped_edges = set((mapping[a], mapping[b]) for a, b in G1.edges())  
        edges2 = set(G2.edges())  
          
        score = len(mapped_edges & edges2)  
          
        if score > best_score:  
            best_score = score  
            best_map = mapping  
      
    return best_map, best_score

import torch.nn as nn  
  
class GraphEncoder(nn.Module):
