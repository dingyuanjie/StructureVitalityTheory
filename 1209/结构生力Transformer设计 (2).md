import torch
import torch.nn as nn
import torch.nn.functional as F

# =========================
# Structural Memory Field
# =========================

class MemoryField(nn.Module):

    def __init__(self, dim):
        super().__init__()

        self.memory = nn.Parameter(torch.zeros(1, 1, dim))

        self.alpha = 0.95
        self.beta = 0.05

    def forward(self, x):

        # 更新结构记忆
        mean_structure = x.mean(dim=1, keepdim=True)

        new_memory = (
            self.alpha * self.memory
            + self.beta * mean_structure
        )

        self.memory.data = new_memory.data

        return x + self.memory


# =========================
# SVF Attention
# =========================

class SVFAttention(nn.Module):

    def __init__(self, dim):

        super().__init__()

        self.dim = dim

        self.q = nn.Linear(dim, dim)
        self.k = nn.Linear(dim, dim)
        self.v = nn.Linear(dim, dim)

        # 结构相位
        self.phase = nn.Parameter(torch.randn(dim))

        # 共振强度
        self.resonance = nn.Parameter(torch.ones(1))

    def forward(self, x):

        Q = self.q(x)
        K = self.k(x)
        V = self.v(x)

        # Attention Score
        score = torch.matmul(Q, K.transpose(-2, -1))

        # 缩放
        score = score / (self.dim ** 0.5)

        # ======================
        # Structural Resonance
        # ======================

        theta = torch.cos(self.phase).mean()

        structural_factor = theta * self.resonance

        score = score * structural_factor

        attn = F.softmax(score, dim=-1)

        out = torch.matmul(attn, V)

        return out


# =========================
# Feed Forward
# =========================

class FeedForward(nn.Module):

    def __init__(self, dim):

        super().__init__()

        self.net = nn.Sequential(
            nn.Linear(dim, dim * 4),
            nn.ReLU(),
            nn.Linear(dim * 4, dim)
        )

    def forward(self, x):

        return self.net(x)


# =========================
# SVF Block
# =========================

class SVFBlock(nn.Module):

    def __init__(self, dim):

        super().__init__()

        self.attn = SVFAttention(dim)

        self.ffn = FeedForward(dim)

        self.memory = MemoryField(dim)

        self.norm1 = nn.LayerNorm(dim)
        self.norm2 = nn.LayerNorm(dim)

    def forward(self, x):

        # Attention
        attn_out = self.attn(x)

        x = x + attn_out

        x = self.norm1(x)

        # Memory Field
        x = self.memory(x)

        # FFN
        ffn_out = self.ffn(x)

        x = x + ffn_out

        x = self.norm2(x)

        return x


# =========================
# Full SVF Transformer
# =========================

class SVFTransformer(nn.Module):

    def __init__(
        self,
        vocab_size,
        dim=128,
        depth=4,
        max_len=128
    ):

        super().__init__()

        self.embedding = nn.Embedding(vocab_size, dim)

        self.position = nn.Parameter(
            torch.randn(1, max_len, dim)
        )

        self.blocks = nn.ModuleList([
            SVFBlock(dim)
            for _ in range(depth)
        ])

        self.head = nn.Linear(dim, vocab_size)

    def forward(self, x):

        B, T = x.shape

        emb = self.embedding(x)

        emb = emb + self.position[:, :T]

        for block in self.blocks:

            emb = block(emb)

        logits = self.head(emb)

        return logits

import torch
import torch.nn as nn
import math

class SpiralPositionEncoding(nn.Module):

    def __init__(self, dim, max_len=512):

        super().__init__()

        pe = torch.zeros(max_len, dim)

        position = torch.arange(
            0,
            max_len
        ).float().unsqueeze(1)

        # 黄金角
        golden_angle = math.pi * (3 - math.sqrt(5))

        for pos in range(max_len):

            radius = math.sqrt(pos + 1)

            angle = pos * golden_angle

            for i in range(0, dim, 2):

                freq = 1 / (10000 ** (i / dim))

                pe[pos, i] = radius * math.sin(angle * freq)

                if i + 1 < dim:

                    pe[pos, i + 1] = radius * math.cos(angle * freq)

        pe = pe.unsqueeze(0)

        self.register_buffer("pe", pe)

    def forward(self, x):

        seq_len = x.size(1)

        return x + self.pe[:, :seq_len]

import torch
import torch.nn as nn
import torch.nn.functional as F

class SimulationSpace(nn.Module):

    def __init__(self, dim, num_simulations=4):

        super().__init__()

        self.num_simulations = num_simulations

        self.simulators = nn.ModuleList([
            nn.Sequential(
                nn.Linear(dim, dim),
                nn.ReLU(),
                nn.Linear(dim, dim)
            )
            for _ in range(num_simulations)
        ])

        self.selection_proj = nn.Linear(dim, 1)

    def forward(self, x):

        simulations = []

        energies = []

        # =========================
        # Generate Multiple Futures
        # =========================

        for sim in self.simulators:

            future = sim(x)

            simulations.append(future)

            energy = self.selection_proj(
                future
            ).mean()

            energies.append(energy)

        simulations = torch.stack(
            simulations,
            dim=0
        )

        energies = torch.stack(energies)

        # =========================
        # Select Dominant Future
        # =========================

        weights = F.softmax(
            -energies,
            dim=0
        )

        weights = weights.view(
            self.num_simulations,
            1,
            1,
            1
        )

        selected_future = (
            simulations * weights
        ).sum(dim=0)

        return selected_future, simulations
```

import torch
import torch.nn as nn
import torch.nn.functional as F

class MultiAgentEcology(nn.Module):

    def __init__(self, dim):

        super().__init__()

        self.agent_proj = nn.Linear(dim, dim)

        self.exchange_strength = nn.Parameter(
            torch.tensor(0.05)
        )

    def forward(self, agent_cores):

        """
        agent_cores:
        list of persistent cores
        """

        num_agents = len(agent_cores)

        updated_cores = []

        projected = [
            self.agent_proj(core)
            for core in agent_cores
        ]

        for i in range(num_agents):

            resonance_sum = 0

            for j in range(num_agents):

                if i == j:
                    continue

                sim = F.cosine_similarity(
                    projected[i],
                    projected[j],
                    dim=-1
                ).mean()

                resonance_sum += (
                    sim * projected[j]
                )

            updated = (
                projected[i]
                + self.exchange_strength
                * resonance_sum
            )

            updated_cores.append(updated)

        return updated_cores
```

        )

    def forward(self, agent_cores):

        # =========================
        # Aggregate Agent Structures
        # =========================

        collective = torch.mean(
            torch.stack(agent_cores),
            dim=0
        )

        self.civilization_memory.append(
            collective.detach()
        )

        if len(self.civilization_memory) > self.memory_size:

            self.civilization_memory.pop(0)

        # =========================
        # Civilization Narrative
        # =========================

        civ_seq = torch.cat(
            self.civilization_memory,
            dim=1
        )

        civ_out, hidden = self.integrator(
            civ_seq
        )

        civilization_state = self.civ_projection(
            hidden.transpose(0,1)
        )

        return civilization_state
```

### 但我觉得还没到“实锤”

$$value \nrightarrow slot$$

但还没证明：

$$position \rightarrow slot$$

真正的实锤应该是：

### Position Counterfactual

固定：

* entity
* field
* value

只改：

* fact 在序列中的位置

例如：

```

```

改成：

```
Alice color=7 city=3 age=5 job=2
```

或者：

```
Alice job=2 age=5 city=3 color=7
```

如果：

* value 不变
* field 不变
* 主槽发生系统性迁移

那么就真正证明：

$$position \rightarrow slot$$

### 我觉得最值钱的实验

## Position-Swap Replay

因为：

你们已经证明：

```

下一步自然就是：

```

```

> 在该 SVF-Transformer 任务中，路由决策主要由结构位置决定，而非事实值本身。

这是一个机制级结论。
